SELECT I.SUMMARY, IL.NAME FROM ISSUES I JOIN ISSUELISTS IL
ORDER BY IL.NAME;

SELECT I.SUMMARY, U.FIRSTNAME, U.LASTNAME FROM ISSUES I JOIN USERS U
ORDER BY U.LASTNAME, U.FIRSTNAME;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_QUANTITY FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 0
ORDER BY U.LASTNAME;
